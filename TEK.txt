  <h1>Berita Sekolah</h1>
  <div id="daftar-berita"></div>

  <script>
    const sheetID = "MASUKKAN_ID_SHEETMU_DISINI";
    const url = `https://docs.google.com/spreadsheets/d/${sheetID}/gviz/tq?tqx=out:json`;

    fetch(url)
      .then(res => res.text())
      .then(rep => {
        const json = JSON.parse(rep.substr(47).slice(0, -2));
        const rows = json.table.rows;
        let html = "";

        rows.forEach(r => {
          const tanggal = r.c[0]?.v || "";
          const judul = r.c[1]?.v || "";
          const isi = r.c[2]?.v || "";
          const gambar = r.c[3]?.v || "";

          html += `
            <div class="berita">
              <div class="judul">${judul}</div>
              <div class="tanggal">${tanggal}</div>
              ${gambar ? `<img src="${gambar}" alt="gambar">` : ""}
              <p>${isi}</p>
            </div>
          `;
        });

        document.getElementById("daftar-berita").innerHTML = html;
      });
  </script>
